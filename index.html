    <script>

        class ScheduleCalendar {

            constructor() {

                this.data = {};

                this.currentEditingBlock = null;

                this.startDate = new Date(2025, 7, 15); // August 15, 2025

                this.endDate = new Date(2025, 10, 16); // November 16, 2025

                this.init();

            }

           

            init() {

                this.generateCalendar();

                this.setupEventListeners();

            }

           

            generateCalendar() {

                const calendarBody = document.getElementById('calendarBody');

                calendarBody.innerHTML = '';

               

                let currentDate = new Date(this.startDate);

               

                while (currentDate <= this.endDate) {

                    const weekStart = this.getWeekStart(currentDate);

                    const weekEnd = new Date(weekStart);

                    weekEnd.setDate(weekEnd.getDate() + 4); // Friday

                   

                    if (weekEnd > this.endDate) {

                        weekEnd.setTime(this.endDate.getTime());

                    }

                   

                    this.generateWeek(weekStart, weekEnd);

                   

                    currentDate = new Date(weekEnd);

                    currentDate.setDate(currentDate.getDate() + 3); // Next Monday

                }

            }

           

            getWeekStart(date) {

                const d = new Date(date);

                const day = d.getDay();

                const diff = d.getDate() - day + (day === 0 ? -6 : 1); // Adjust for Monday start

                return new Date(d.setDate(diff));

            }

           

            generateWeek(weekStart, weekEnd) {

                const calendarBody = document.getElementById('calendarBody');

                const weekContainer = document.createElement('div');

                weekContainer.className = 'week-container';

               

                const weekHeader = document.createElement('div');

                weekHeader.className = 'week-header';

                weekHeader.textContent = `Week of ${this.formatDate(weekStart)}`;

                weekContainer.appendChild(weekHeader);

               

                // Add each day as a separate row

                const days = ['Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday'];

                for (let i = 0; i < 5; i++) {

                    const currentDay = new Date(weekStart);

                    currentDay.setDate(weekStart.getDate() + i);

                   

                    if (currentDay >= this.startDate && currentDay <= this.endDate) {

                        const dayGrid = this.createDayGrid(days[i], currentDay, i === 4);

                        weekContainer.appendChild(dayGrid);

                    }

                }

               

                calendarBody.appendChild(weekContainer);

            }

           

            createDayGrid(dayName, date, isFriday) {

                const dayGrid = document.createElement('div');

                dayGrid.className = 'day-grid';

               

                // Day label

                const dayLabel = document.createElement('div');

                dayLabel.className = 'day-label';

                dayLabel.innerHTML = `

                    <div class="day-name">${dayName}</div>

                    <div class="day-date">${date.getDate()}</div>

                `;

                dayGrid.appendChild(dayLabel);

               

                if (isFriday) {

                    // Friday has 3 blocks

                    const fridaySlots = [

                        { label: '12-4pm', start: 12, end: 16 },

                        { label: '4-8pm', start: 16, end: 20 },

                        { label: '3-6pm', start: 15, end: 18, overlap: true }

                    ];

                   

                    fridaySlots.forEach(slot => {

                        const block = this.createTimeBlock(date, slot, slot.overlap ? 'friday-overlap' : 'friday');

                        dayGrid.appendChild(block);

                    });

                } else {

                    // Regular weekdays have 2 blocks

                    const weekdaySlots = [

                        { label: '12-3pm', start: 12, end: 15 },

                        { label: '3-6pm', start: 15, end: 18 }

                    ];

                   

                    weekdaySlots.forEach(slot => {

                        const block = this.createTimeBlock(date, slot, 'weekday');

                        dayGrid.appendChild(block);

                    });

                }

               

                return dayGrid;

            }

           

            // Remove the old addDayRow function since we're using createDayGrid now

           

            createTimeBlock(date, slot, type) {

                const block = document.createElement('div');

                block.className = 'time-block';

               

                if (type === 'friday-overlap') {

                    block.classList.add('friday-overlap');

                }

               

                const blockId = `${date.getFullYear()}-${date.getMonth()}-${date.getDate()}-${slot.start}-${slot.end}${slot.overlap ? '-overlap' : ''}`;

                block.dataset.blockId = blockId;

               

                const content = document.createElement('div');

                content.className = 'block-content';

                content.textContent = this.data[blockId] || '';

               

                if (this.data[blockId]) {

                    block.classList.add('has-content');

                }

               

                block.appendChild(content);

               

                block.addEventListener('click', () => {

                    this.openEditModal(blockId, date, slot);

                });

               

                return block;

            }

           

            createUnavailableBlock() {

                const block = document.createElement('div');

                block.className = 'time-block unavailable';

                return block;

            }

           

            openEditModal(blockId, date, slot) {

                this.currentEditingBlock = blockId;

                const modal = document.getElementById('editModal');

                const modalTitle = document.getElementById('modalTitle');

                const modalTextarea = document.getElementById('modalTextarea');

               

                const dayName = date.toLocaleDateString('en-US', { weekday: 'long' });

                const dateStr = date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });

               

                modalTitle.textContent = `${dayName}, ${dateStr} - ${slot.label}`;

                modalTextarea.value = this.data[blockId] || '';

               

                modal.style.display = 'block';

                modalTextarea.focus();

            }

           

            setupEventListeners() {

                const modal = document.getElementById('editModal');

                const saveBtn = document.getElementById('saveBtn');

                const cancelBtn = document.getElementById('cancelBtn');
